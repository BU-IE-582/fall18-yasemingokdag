train_data <- cbind(train_x, train_y[,2:ncol(train_y)], train_z[,2:ncol(train_z)])
train_data <- as.data.table(train_data)

#Check counts per data 
class_count <- train_data[, .(.N), by = .(Class)]

#We want to have equal number of data points from each class per fold (startification)
require(TunePareto)


#
k_levels=c(1,2,3,4,5)
thresHold=0.5
nofReplications=1
nFolds=10
indices=generateCVRuns(t(train_data[,1]),nofReplications,nFolds,stratified=TRUE)

cvresult=data.table()

for(i in 1:nFolds)
{
  testindices=indices[[1]][[i]]
  cvtrain=train_data[-testindices,]        
  cvtest=train_data[testindices,]
  
  for(y in 1:length(k_levels)){
    param_k <- k_levels[y]
    predict <-  kknn(as.factor(Class) ~ .,
                     cvtrain,
                     cvtest,
                     k = param_k,
                     distance = 2,
                     kernel = "rectangular",
                     scale = FALSE)
    cvresult=rbind(cvresult,data.table(Fold=i,Method='knn',Klev=param_k,TestId = testindices,
                                       Predictions=predict$fitted.value,Real=cvtest[,1]))
    print(paste0('fold ', i, ' for k ', param_k, ' completed'))
    
  }   
}    

#Manhattan

manh_cvresult=data.table()
for(i in 1:nFolds)
{

}
